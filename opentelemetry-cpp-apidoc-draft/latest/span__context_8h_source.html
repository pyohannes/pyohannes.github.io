27 /* SpanContext contains the state that must propagate to child Spans and across
28  * process boundaries. It contains the identifiers TraceId and SpanId,
29  * TraceFlags, TraceState, and whether it was propagated from a remote parent.
30  *
31  * TODO: This is currently a placeholder class and requires revisiting
32  */
33 class SpanContext final
34 {
35 public:
36  /* A temporary constructor for an invalid SpanContext.
37  * Trace id and span id are set to invalid (all zeros).
38  *
39  * @param sampled_flag a required parameter specifying if child spans should be
40  * sampled
41  * @param is_remote true if this context was propagated from a remote parent.
42  */
43  SpanContext(bool sampled_flag, bool is_remote)
44  : trace_id_(),
45  span_id_(),
46  trace_flags_(trace_api::TraceFlags((uint8_t)sampled_flag)),
47  is_remote_(is_remote)
48  {}
49 
50  // @returns whether this context is valid
51  bool IsValid() const noexcept { return trace_id_.IsValid() && span_id_.IsValid(); }
52 
53  // @returns the trace_flags associated with this span_context
54  const trace_api::TraceFlags &trace_flags() const noexcept { return trace_flags_; }
55 
56  const trace_api::TraceId &trace_id() const noexcept { return trace_id_; }
57 
58  const trace_api::SpanId &span_id() const noexcept { return span_id_; }
59 
60  SpanContext(TraceId trace_id, SpanId span_id, TraceFlags trace_flags, bool is_remote) noexcept
61  : trace_id_(trace_id), span_id_(span_id), trace_flags_(trace_flags), is_remote_(is_remote)
62  {}
63 
64  SpanContext(const SpanContext &ctx) = default;
65 
66  SpanContext &operator=(const SpanContext &ctx) = default;
67 
68  bool operator==(const SpanContext &that) const noexcept
69  {
70  return trace_id() == that.trace_id() && span_id() == that.span_id() &&
71  trace_flags() == that.trace_flags();
72  }
73 
74  bool IsRemote() const noexcept { return is_remote_; }
75 
76  static SpanContext GetInvalid() { return SpanContext(false, false); }
77 
78  bool IsSampled() const noexcept { return trace_flags_.IsSampled(); }
79 
80 private:
81  trace_api::TraceId trace_id_;
82  trace_api::SpanId span_id_;
83  trace_api::TraceFlags trace_flags_;
84  bool is_remote_ = false;
85 };
86 } // namespace trace
